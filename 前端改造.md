|--------|---------|---------|
| LLM API 调用 | 完全依赖 ST 代理 | 无直接 API 代码 |

### 1.3 改造目标

**让 index.html 脱离 SillyTavern，以独立网页运行，直接对接 LLM API。**

部署形态：
- 静态托管（GitHub Pages / Vercel），用户自带 API Key
- 打包为 APK（Capacitor）或 EXE（Tauri），用户自带 API Key

设计原则：
- 最小改动实现独立化，不做不必要的重构
- 保持现有全局函数式风格，仅新增必要服务模块
- 保留 ST 兼容性（双模式分支）

```javascript
// api/api-service.js

class APIService {
    constructor() {
        this.config = {
            endpoint: '',
            apiKey: '',
            model: '',
            type: 'openai',     // 'openai' | 'gemini'

            maxOutputTokens: 8192,

        };
    }

    loadConfig() {
        const saved = localStorage.getItem('jxz_apiConfig');
        if (saved) Object.assign(this.config, JSON.parse(saved));
    }

    saveConfig() {
        localStorage.setItem('jxz_apiConfig', JSON.stringify(this.config));
    }

    async healthCheck() {
        // 发一条极短消息验证连通性
    }

    /** 非流式调用 */
    async sendMessages(messages) {
        if (this.config.type === 'gemini') {
            return this._callGemini(messages);
        }
        return this._callOpenAI(messages);
    }

    /** 流式调用 */
    async sendMessagesStream(messages, onStateChange) {
        const response = await fetch(`${this.config.endpoint}/chat/completions`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.config.apiKey}`
            },
            body: JSON.stringify({
                model: this.config.model,
                messages,
                temperature: this.config.temperature,
                max_tokens: this.config.maxOutputTokens,
                stream: true,
            })
        });

        if (!response.ok) {
            throw new Error(`API 错误 ${response.status}: ${await response.text()}`);
        }

        return processSSEStream(response, onStateChange);
    }

    async _callOpenAI(messages) {
        const response = await fetch(`${this.config.endpoint}/chat/completions`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.config.apiKey}`
            },
            body: JSON.stringify({
                model: this.config.model,
                messages,
                temperature: this.config.temperature,
                max_tokens: this.config.maxOutputTokens,
            })
        });
        if (!response.ok) throw new Error(`API 错误 ${response.status}`);
        const data = await response.json();
        return {
            content: data.choices[0].message.content,
            usage: data.usage,
        };
    }

    async _callGemini(messages) {
        const systemMsgs = messages.filter(m => m.role === 'system');
        const chatMsgs = messages.filter(m => m.role !== 'system');
        const systemPrompt = systemMsgs.map(m => m.content).join('\n\n');

        const url = `${this.config.endpoint}/models/${this.config.model}:generateContent?key=${this.config.apiKey}`;
        const contents = chatMsgs.map(m => ({
            role: m.role === 'assistant' ? 'model' : 'user',
            parts: [{ text: m.content }]
        }));

        const response = await fetch(url, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                contents,
                systemInstruction: { parts: [{ text: systemPrompt }] },
                generationConfig: {
                    temperature: this.config.temperature,
                    maxOutputTokens: this.config.maxOutputTokens,
                }
            })
        });
        if (!response.ok) throw new Error(`Gemini 错误 ${response.status}`);
        const data = await response.json();
        return {
            content: data.candidates[0].content.parts[0].text,
            usage: data.usageMetadata,
        };
    }
}
```

### 3.2 流式 SSE 处理（reducer 模式）

借鉴 TavernHeadless `client-helpers/stream/reduce-respond-stream.ts` 的纯函数 reducer 模式：

```javascript
// api/stream-handler.js

function createInitialStreamState() {
    return { status: 'idle', content: '', error: null, usage: null };
}

function reduceStreamEvent(state, event) {
    switch (event.type) {
        case 'start':
            return { ...state, status: 'streaming' };
        case 'chunk':
            return { ...state, content: state.content + event.text, status: 'streaming' };
        case 'done':
            return { ...state, usage: event.usage ?? null, status: 'done' };
        case 'error':
            return { ...state, error: { message: event.message }, status: 'error' };
        default:
            return state;
    }
}

async function processSSEStream(response, onStateChange) {
    let state = createInitialStreamState();
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';

    onStateChange(state = reduceStreamEvent(state, { type: 'start' }));

    while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop();

        for (const line of lines) {
            if (!line.startsWith('data: ')) continue;
            const data = line.slice(6);
            if (data === '[DONE]') {
                onStateChange(state = reduceStreamEvent(state, { type: 'done' }));
                return state;
            }
            try {
                const parsed = JSON.parse(data);
                const text = parsed.choices?.[0]?.delta?.content ?? '';
                if (text) {
                    onStateChange(state = reduceStreamEvent(state, { type: 'chunk', text }));
                }
            } catch { /* 忽略非 JSON 行 */ }
        }
    }

    onStateChange(state = reduceStreamEvent(state, { type: 'done' }));
    return state;
}
```

### 3.3 模板引擎

借鉴 TavernHeadless `core/prompt/template-engine.ts`，兼容 ST 的 `{{变量}}` 语法：

```javascript
// api/template-engine.js

const TEMPLATE_REGEX = /\{\{\s*([^{}:]+?)\s*(?::\s*([^{}]*?)\s*)?\}\}/g;

/**
 * 将模板中的 {{var}} 替换为实际值，支持 {{var:默认值}} 语法
 */
function renderTemplate(template, variables, undefinedBehavior = 'keep') {
    return template.replace(TEMPLATE_REGEX, (match, name, defaultValue) => {
        const key = name.trim();
        if (key in variables) {
            const value = variables[key];
            if (value === null || value === undefined) return '';
            if (typeof value === 'object') return JSON.stringify(value);
            return String(value);
        }
        if (defaultValue !== undefined) return defaultValue.trim();
        return undefinedBehavior === 'empty' ? '' : match;
    });
}

function extractVariableNames(template) {
    const names = new Set();
    const regex = new RegExp(TEMPLATE_REGEX.source, TEMPLATE_REGEX.flags);
    let match;
    while ((match = regex.exec(template)) !== null) names.add(match[1].trim());
    return [...names];
}
```

### 3.4 世界书触发引擎

借鉴 TavernHeadless `adapters-sillytavern/worldbook/trigger-engine.ts`，实现按需注入 NPC/地点信息：

```javascript
// api/worldbook-trigger.js

/**
 * @typedef {Object} WorldBookEntry
 * @property {string} id
 * @property {string} content - 条目内容（可含 {{变量}}）
 * @property {string[]} keys - 触发关键词（支持 /正则/i 格式）
 * @property {string[]} [secondaryKeys] - 次要关键词（需同时命中）
 * @property {'before'|'after'|'depth'} position - 注入位置
 * @property {number} [depth] - depth 插入深度
 * @property {boolean} [constant] - 常驻注入
 * @property {number} [scanDepth] - 扫描最近 N 条消息
 * @property {number} [order] - 排序权重（大的靠前）
 */

class WorldBookTriggerEngine {
    constructor(entries) { this.entries = entries; }

    /**
     * 扫描消息，确定哪些条目需要激活
     * @returns {{ before: Entry[], after: Entry[], atDepth: {entry, depth}[] }}
     */
    trigger(recentMessages, defaultScanDepth = 10) {
        const haystack = recentMessages.slice(0, defaultScanDepth).join('\n');
        const activated = [];

        for (const entry of this.entries) {
            if (entry.constant) { activated.push(entry); continue; }

            const scanText = entry.scanDepth
                ? recentMessages.slice(0, entry.scanDepth).join('\n')
                : haystack;

            if (!this._matchKeys(entry.keys, scanText, entry.caseSensitive)) continue;
            if (entry.secondaryKeys?.length > 0 &&
                !this._matchKeys(entry.secondaryKeys, scanText, entry.caseSensitive)) continue;

            activated.push(entry);
        }

        activated.sort((a, b) => (b.order ?? 0) - (a.order ?? 0));

        return {
            before: activated.filter(e => e.position === 'before'),
            after: activated.filter(e => e.position === 'after'),
            atDepth: activated.filter(e => e.position === 'depth').map(e => ({ entry: e, depth: e.depth ?? 0 })),
        };
    }

    _matchKeys(keys, text, caseSensitive = false) {
        for (const key of keys) {
            if (key.startsWith('/') && key.lastIndexOf('/') > 0) {
                try {
                    const lastSlash = key.lastIndexOf('/');
                    const regex = new RegExp(key.slice(1, lastSlash), key.slice(lastSlash + 1) || 'i');
                    if (regex.test(text)) return true;
                } catch { /* skip */ }
            } else {
                const a = caseSensitive ? text : text.toLowerCase();
                const b = caseSensitive ? key : key.toLowerCase();
                if (a.includes(b)) return true;
            }
        }
        return false;
    }
}
```

**加载策略（分阶段）：**

- **Phase 1（不使用世界书）**：将 5 个核心 system prompt 文件硬编码为 JS 常量（`prompt-data-core.js`），其余不加载。首屏零 fetch 延迟。
- **Phase 2（世界书启用）**：启动时并行 `Promise.all(fetch(...))` 加载全部 txt 文件，写入 IndexedDB 缓存。后续启动优先读缓存，仅在版本号变化时重新 fetch。
- **可选构建优化**：提供 `scripts/build-prompt-data.js` 脚本，将全部 txt 编译为单文件 `prompt-data-all.js`（~150KB），适合打包 APK/EXE 的场景。

**Phase 1 核心文件固定为：**

- `010天山派背景.txt`
- `020地点介绍.txt`
- `040主角属性.txt`
- `105列表.txt`
- `110格式规范_精简版.txt`

选择理由：

1. `010` 提供稳定的世界观和门派背景，是所有剧情生成的基础约束
2. `020` 提供当前地点舞台信息，直接影响场景描写与 NPC 行动合理性
3. `040` 承载主角属性、资质、出身等动态分支，是角色互动成立的前提
4. `105` 提供 SIDE_NOTE 与 SLG 输出依赖的枚举列表，能减少模型输出漂移
5. `110` 定义最终输出格式，是解析链路成立的硬约束

**不纳入 Phase 1 system prompt 的文件：**

- `050开场白.txt`：不作为 system prompt 常驻，而是在新游戏开始后作为首条 `assistant` 历史消息写入前端页面与对话历史

这样区分的原因是：`050` 属于“初始剧情内容”，不是“持续约束模型的规则文本”。将其放入历史而不是 system prompt，更符合现有 ST 卡片的语义，也便于后续重新开始游戏时复用。

### 3.5 Prompt IR 与 MessageBuilder

借鉴 TavernHeadless `core/prompt/message-builder.ts` + `token-budget.ts`，使用中间表示（IR）统一管理多来源提示词的顺序和裁剪：

```javascript
// api/prompt-builder.js

/**
 * Prompt IR 数据结构：
 * - IRMessage: { role, content, source, tokenCount, prunable, priority }
 * - IRSection: { name, messages[], pinned, budgetGroup, order, insertionKind, depth }
 * - PromptIR:  { sections[], metadata: { maxTokens, reservedForReply } }
 *
 * priority 规则：值越大 → 越先被裁剪（旧消息 priority 高）
 */

class MessageBuilder {
    constructor(maxTokens, reservedForReply) {
        this.maxTokens = maxTokens;
        this.reservedForReply = reservedForReply;
    }

    /** 完整流程：estimate → prune → assemble */
    build(ir) {
        const estimated = this._estimate(ir);
        const pruned = this._prune(estimated);
        return this._assemble(pruned);
    }

    _estimate(ir) {
        return { ...ir, sections: ir.sections.map(s => ({
            ...s, messages: s.messages.map(m => ({
                ...m, tokenCount: m.tokenCount ?? this._countTokens(m.content)
            }))
        }))};
    }

    _prune(ir) {
        const budget = this.maxTokens - this.reservedForReply;
        let fixedTokens = 0;
        const prunableMsgs = [];

        for (const section of ir.sections) {
            for (const msg of section.messages) {
                if (section.pinned || msg.prunable === false) {
                    fixedTokens += msg.tokenCount;
                } else {
                    prunableMsgs.push(msg);
                }
            }
        }

        // priority 大的先裁剪（旧历史消息 priority 高）
        prunableMsgs.sort((a, b) => (b.priority ?? 0) - (a.priority ?? 0));

        const kept = new Set();
        let remaining = budget - fixedTokens;
        // 从 priority 小的（重要的）开始保留
        for (const msg of [...prunableMsgs].reverse()) {
            if (remaining >= msg.tokenCount) {
                kept.add(msg);
                remaining -= msg.tokenCount;
            }
        }

        return { ...ir, sections: ir.sections.map(s => ({
            ...s, messages: s.messages.filter(m =>
                s.pinned || m.prunable === false || kept.has(m))
        }))};
    }

    _assemble(ir) {
        const sorted = [...ir.sections].sort((a, b) => a.order - b.order);
        const messages = [];
        for (const section of sorted) {
            if (section.insertionKind === 'in_chat') continue;
            for (const msg of section.messages) {
                messages.push({ role: msg.role, content: msg.content });
            }
        }
        return messages;
    }

    /**
     * Token 估算：中文 ≈ 1.5 token/字，英文 ≈ 0.25 token/字符
     * 含 10% 安全余量，避免标点/数字混排导致的低估
     *
     * Phase 3 可替换为 tiktoken-lite 精确计数
     */
    _countTokens(text) {
        const cjk = (text.match(/[\u4e00-\u9fff\u3000-\u303f]/g) || []).length;
        const raw = Math.ceil(cjk * 1.5 + (text.length - cjk) / 4);
        return Math.ceil(raw * 1.1);  // 10% 安全余量
    }
}

class PromptBuilder {
    constructor(config) {
        this.config = config;
        this.worldbook = null;
        this.messageBuilder = null;
        this.systemPromptParts = [];
    }

    async init() {
        this.messageBuilder = new MessageBuilder(
            this.config.maxContextTokens || 128000,
            this.config.maxOutputTokens || 8192
        );
        const entries = await this._loadWorldBookEntries();
        this.worldbook = new WorldBookTriggerEngine(entries);
        await this._loadSystemPrompt();
    }

    /**
     * 构建完整的 messages[] — 核心方法
     * 通过 IR 统一管理顺序、预算和裁剪
     */
    buildMessages({ userMessage, gameData, conversationHistory, memories = [] }) {
        const variables = { user: gameData.playerName || '主角', ...this._flattenGameData(gameData) };
        const sections = [];

        // 1. System Prompt（pinned）
        sections.push({
            name: 'system_prompt', order: 0, pinned: true,
            messages: [{ role: 'system', content: this._buildSystemPrompt(variables), prunable: false }]
        });

        // 2. 世界书 before（关键词触发，pinned）
        const recentTexts = conversationHistory.map(m => m.content).concat(userMessage);
        const triggered = this.worldbook.trigger(recentTexts);
        if (triggered.before.length > 0) {
            sections.push({
                name: 'worldbook_before', order: 1, pinned: true, budgetGroup: 'worldbook',
                messages: triggered.before.map(e => ({
                    role: 'system', content: renderTemplate(e.content, variables), prunable: false,
                }))
            });
        }

        // 3. 游戏状态快照（pinned）
        sections.push({
            name: 'state_snapshot', order: 2, pinned: true,
            messages: [{ role: 'system', content: this._buildStateContext(gameData), prunable: false }]
        });

        // 4. 远期记忆（可裁剪，低 priority = 优先保留）
        if (memories.length > 0) {
            sections.push({
                name: 'memory', order: 3, pinned: false, budgetGroup: 'memory',
                messages: [{ role: 'system', content: this._formatMemories(memories), prunable: true, priority: 1 }]
            });
        }

        // 5. 对话历史（可裁剪，旧消息 priority 高 → 优先丢弃）
        sections.push({
            name: 'chat_history', order: 4, pinned: false, budgetGroup: 'history',
            messages: conversationHistory.map((msg, i) => ({
                role: msg.role, content: msg.content, prunable: true,
                priority: conversationHistory.length - i,  // 旧的 priority 大，先裁
            }))
        });

        // 6. 世界书 after（pinned）
        if (triggered.after.length > 0) {
            sections.push({
                name: 'worldbook_after', order: 5, pinned: true, budgetGroup: 'worldbook',
                messages: triggered.after.map(e => ({
                    role: 'system', content: renderTemplate(e.content, variables), prunable: false,
                }))
            });
        }

        // 7. 当前用户输入（pinned）
        sections.push({
            name: 'user_input', order: 6, pinned: true,
            messages: [{ role: 'user', content: userMessage, prunable: false }]
        });

        // 通过 MessageBuilder 裁剪并输出
        const ir = { sections, metadata: { maxTokens: this.config.maxContextTokens, reservedForReply: this.config.maxOutputTokens } };
        return this.messageBuilder.build(ir);
    }

    async _loadWorldBookEntries() { /* fetch char_card_information/ 下的文件并转为条目 */ }
    async _loadSystemPrompt() { /* Phase 1 加载 010/020/040/105/110；Phase 2 再扩展 */ }
    _buildSystemPrompt(variables) { return renderTemplate(this.systemPromptParts.join('\n\n'), variables); }
    _buildStateContext(gameData) { /* 格式化当前游戏状态为文本 */ }
    _formatMemories(memories) { return '【远期记忆】\n' + memories.map(m => `- ${m.content}`).join('\n'); }
    _flattenGameData(gameData) { return { currentWeek: gameData.currentWeek, location: gameData.location /*, ...*/ }; }
}
```

#### 3.5.1 Phase 1 的核心 Prompt 组成

Phase 1 的 `system prompt` 固定由以下 5 段组成，按顺序拼接：

```javascript
// api/prompt-data-core.js
const CORE_PROMPT_FILES = [
    '010天山派背景',
    '020地点介绍',
    '040主角属性',
    '105列表',
```

推荐顺序：

1. 背景世界观
2. 当前地点信息
3. 主角个人信息
4. 枚举列表
5. 输出格式规范

原因是模型会先建立“世界与人物”，再读到“可选枚举”，最后读到“严格输出格式”，约束效果更稳定。

#### 3.5.2 开场白的处理

`050开场白.txt` 不进入 system prompt，而是在“新游戏初始化成功”后自动注入为首条 `assistant` 历史消息：

```javascript
// runtime/app-runtime.js
async function ensureOpeningMessage() {
    const history = await storageService.loadConversationHistory();
    if (history.length > 0) return;

    const openingText = renderOpeningText(gameData);
    const openingMsg = {
        id: crypto.randomUUID(),
        role: 'assistant',
        content: openingText,
        kind: 'opening',
        createdAt: Date.now(),
    };

    await storageService.appendConversation(openingMsg);
    renderMainText(openingText);

```

约定如下：

1. 仅在“新游戏且历史为空”时注入一次
2. 注入后立即渲染到首页故事区域
3. 后续构建 prompt 时，它和普通历史消息一样参与 `conversationHistory`
4. 读档时若存档中已有历史，则不重复注入

这能保证开场白既出现在页面上，又自然成为后续剧情延续的历史上下文。

#### 3.5.3 EJS / 变量分支的实现策略

当前提示词文件并不只是简单的 `{{变量}}` 替换，而是混合了两类动态能力：

1. `{{user}}` 这类占位符替换
2. `<% if (...) { %> ... <% } %>` 这类条件分支与局部变量定义

从现有文件看，Phase 1 至少有以下典型分支：

- `020地点介绍.txt`：根据 `gameData.GameMode`、`gameData.mapLocation` 选择地点块
- `040主角属性.txt`：根据人物数值区间和 `userBackground` 输出不同描述
- `105列表.txt`：根据 `gameData.GameMode` 切换列表枚举
- `110格式规范_精简版.txt`：根据 `newWeek`、`GameMode`、`randomEvent`、`battleEvent` 切换输出格式

**Phase 1 不建议直接在浏览器引入完整 EJS。** 原因不是功能做不到，而是会引入过大的模板运行时和不必要的自由度，增加调试成本。更合适的做法是实现一个“EJS 子集编译器”，只支持当前文件实际使用到的语法。

支持范围限定为：

```javascript
// api/template-engine.js
// Phase 1 支持的 EJS 子集：
// 1. <%_ const foo = expr; _%>
// 2. <%_ if (expr) { _%>
// 3. <%_ } else if (expr) { _%>
// 4. <%_ } else { _%>
// 5. <%_ } _%>
// 6. <%- expr %> / <%= expr %>
// 7. 最终再做 {{var}} 替换
```

实现流程：

```javascript
function renderPromptTemplate(template, context) {
    const ejsRendered = renderEJSLite(template, context);
    return renderTemplate(ejsRendered, context, 'empty');
}
```

`renderEJSLite` 的本质是把模板编译成一个受控函数：

```javascript
function compileEJSLite(template) {
    const source = templateToFunctionSource(template);
    return new Function('ctx', `
        const { gameData, user } = ctx;
        let __out = '';
        ${source}
        return __out;
    `);
}
```

但必须加两层约束：

1. **只允许读取传入的 `ctx`**，不暴露 `window`、`document`、`fetch` 等浏览器对象
2. **只服务仓库内自带 prompt 文件**，不执行用户输入的任意模板

如果后续实现时想进一步收敛风险，还可以不做通用编译器，而是做“预编译脚本”：

- 开发期运行 `scripts/build-prompt-data.js`
- 将 010/020/040/105/110 编译成 `getPromptCore(gameData)` 形式的 JS
- 浏览器端只消费编译结果，不再运行模板解释器

这两种方案里，**我建议 Phase 1 先做运行时 `EJSLite`，Phase 2 或打包场景再做预编译优化**。原因是：

1. 当前模板规则比较固定，运行时实现工作量小
2. 先保真迁移现有卡片逻辑，便于对照 ST 行为排查问题
3. 等独立前端跑通后，再把热点模板编译成静态函数即可

#### 3.5.4 Phase 1 的历史保留策略

Phase 1 不再把“完整对话原文”作为主要上下文来源，而是采用**摘要历史优先**策略：

1. 每轮 AI 回复解析出 `<SUMMARY>` 后，只把 `SUMMARY` 内容写入 `summaryHistory`
2. 下一轮构建 prompt 时，不回灌旧的 `<MAIN_TEXT>` 原文，只注入 `summaryHistory`
3. 当前轮的用户输入仍以 `user` 消息单独传入，不受历史裁剪影响
4. 超出上下文窗口时，直接丢弃**最旧的 `<SUMMARY>`**，Phase 1 不做合并压缩，也不做向量召回

也就是说，Phase 1 的 prompt 上下文分为两层：

- **UI 历史**：用于前端页面展示、刷新恢复、存档预览，可保留开场白和正文原文
- **Prompt 历史**：仅由历轮 `<SUMMARY>` 构成，专门服务模型上下文

PromptBuilder 在 Phase 1 中应改为优先消费 `summaryHistory`：

```javascript
buildMessages({ userMessage, gameData, summaryHistory = [] }) {
    const variables = { user: gameData.playerName || '主角', ...this._flattenGameData(gameData) };
    const messages = [];

    messages.push({
        role: 'system',
        content: this._buildSystemPrompt(variables),
    });

    const recentHistoryBlock = this._buildRecentHistoryBlock(summaryHistory);
    if (recentHistoryBlock) {
        messages.push({ role: 'system', content: recentHistoryBlock });
    }

    messages.push({
        role: 'system',
        content: this._buildStateContext(gameData),
    });

    messages.push({ role: 'user', content: userMessage });
    return messages;
}
```

其中 `<RecentHistory>` 的来源直接对应 [char_card_information/110格式规范_精简版.txt](char_card_information/110格式规范_精简版.txt) 中要求模型在特定分支下回顾的 `<RecentHistory>` 标签：

```javascript
_buildRecentHistoryBlock(summaryHistory) {
    if (!summaryHistory || summaryHistory.length === 0) return '';

    return [
        '<RecentHistory>',
        ...summaryHistory.map(item => item.summaryText),
        '</RecentHistory>',
    ].join('\n');
}
```

**截断规则（Phase 1 固定策略）：**

```javascript
function selectSummaryHistory(summaryRecords, maxTokens) {
    const selected = [];
    let used = 0;

    for (const item of [...summaryRecords].reverse()) {
        const tokens = fastEstimateTokens(item.summaryText);
        if (used + tokens > maxTokens) break;
        selected.push(item);
        used += tokens;
    }

    return selected.reverse();
}
```

规则说明：

1. 始终从**最新摘要**开始向前回收
2. 一旦超过预算，直接停止，不再保留更老摘要
3. 被窗口排除的旧摘要不进入本轮 prompt，等 Phase 2 再考虑向量化召回
4. 这是**滑动窗口 + 丢弃旧摘要**，不是压缩、重写或再摘要

建议 Phase 1 先使用硬上限：

- `maxSummaryHistoryItems = 20`
- `maxSummaryHistoryTokens = min(2000, floor(maxContextTokens * 0.15))`

这样做的原因：

1. `110格式规范_精简版.txt` 在 `newWeek != 0` 的分支里已经要求模型输出“按条罗列的简短 SUMMARY”，天然适合做摘要历史
2. 相比回灌旧正文，`SUMMARY` 更稳定、更省 token，也更容易控制窗口
3. Phase 1 先明确“只保留摘要，不保留旧正文”，后续升级向量召回时边界更清晰

**开场白与摘要历史的关系：**

`050开场白.txt` 仍按上一节方案作为 UI 首条 `assistant` 消息写入页面，但**默认不进入 `summaryHistory`**。如果实测发现首轮衔接不足，再补一个固定的 `openingSummary` 条目即可；Phase 1 先不引入这层额外复杂度。

### 3.6 三级变量系统

变量系统不再推迟到 Phase 2，而是**Phase 1 就落地**。原因是当前项目里 `gameData` 已经承载了绝大多数状态字段，而且 `summary_Small` / `summary_Week` / `summary_Backup` 也已经是现有存档结构的一部分。现在就引入 `variableSystem`，可以把“运行时变量读写”“重新生成回滚”“独立前端存档”和“兼容现有字段”统一起来。

**先确认现状：** 现有代码中，`summary_Small` / `summary_Week` / `summary_Backup` 确实已经在 `defaultGameData`、运行时变量和 `syncGameDataFromVariables()` / `syncVariablesFromGameData()` 中双向同步，并随 `saveGameData()` 一起持久化。也就是说，它们不是临时变量，而是现有存档协议的一部分。

因此 Phase 1 的策略不是删除它们，而是：

1. `variableSystem.sessionVars` 以 `gameData` 为底层快照来源
2. `summaryHistory` 作为独立数组保存逐条摘要，供 PromptBuilder 生成 `<RecentHistory>`
3. `summary_Small` / `summary_Week` / `summary_Backup` 继续保留在 `gameData` 中，作为**兼容现有逻辑和旧存档的聚合摘要层**
4. 存档时同时保存 `gameData` 和 `summaryHistory`，加载时再共同恢复 prompt 上下文

借鉴 TavernHeadless `core/variables/variable-resolver.ts` 的级联读取思路，精简为前端适用的 3 级：

```javascript
// storage/variable-system.js

/**
 * turn:    当前回合（重新生成时回滚）
 * session: 当前存档（= gameData，跨回合持久）
 * global:  全局设置（跨存档持久）
 *
 * 读取优先级：turn → session → global
 */
class VariableSystem {
    constructor(initialSession = {}) {
        this.turnVars = new Map();
        this.sessionVars = new Map(Object.entries(initialSession));
        this.globalVars = new Map();
    }

    get(key) {
        if (this.turnVars.has(key)) return this.turnVars.get(key);
        if (this.sessionVars.has(key)) return this.sessionVars.get(key);
        return this.globalVars.get(key);
    }

    set(key, value, scope = 'turn') {
        this[`${scope}Vars`].set(key, value);
    }

    /** 回合确认成功：turn 变量提升到 session */
    commitTurn() {
        for (const [k, v] of this.turnVars) this.sessionVars.set(k, v);
        this.turnVars.clear();
    }

    /** 重新生成：丢弃 turn 变量 */
    rollbackTurn() { this.turnVars.clear(); }

    /** 存档时导出 session 快照 */
    exportSession() { return Object.fromEntries(this.sessionVars); }

    /** 读档时导入 */
    importSession(snapshot) {
        this.sessionVars = new Map(Object.entries(snapshot));
        this.turnVars.clear();
    }

    /** 供模板引擎使用的合并变量 */
    toTemplateVars() {
        const merged = new Map(this.globalVars);
        for (const [k, v] of this.sessionVars) merged.set(k, v);
        for (const [k, v] of this.turnVars) merged.set(k, v);
        return Object.fromEntries(merged);
    }
}
```

#### 3.6.1 `variableSystem` 与 `gameData` 的关系

Phase 1 明确采用“**变量系统包裹 gameData，而不是替代 gameData**”的方案：

```javascript
// runtime/app-runtime.js
gameData = await storageService.loadGameDataOrDefault();
variableSystem = new VariableSystem(gameData);
```

职责划分如下：

| 层 | 角色 | 用途 |
|----|------|------|
| `gameData` | 存档快照 / 兼容层 | 持久化所有游戏字段，兼容现有模块和旧存档 |
| `variableSystem.sessionVars` | 当前存档运行态 | 运行时主读取源，跨回合持久 |
| `variableSystem.turnVars` | 本回合暂存 | API 成功后提交、重新生成时回滚 |
| `summaryHistory` | Prompt 摘要历史 | 逐条 `<SUMMARY>`，专供 `<RecentHistory>` 注入 |

这意味着：

1. 页面和旧模块仍然可以继续读写 `gameData.xxx`
2. 新增独立前端流水线优先读写 `variableSystem`
3. 每次提交成功后，把 `variableSystem.exportSession()` 同步回 `gameData`
4. 存档时最终写盘的是 `gameData + summaryHistory`

同步方式建议如下：

```javascript
function syncGameDataFromVariableSystem() {
    const sessionSnapshot = variableSystem.exportSession();
    gameData = mergeWithDefaults(sessionSnapshot, defaultGameData);
}

function hydrateVariableSystemFromGameData(gameData) {
    variableSystem = new VariableSystem(gameData);
}
```

#### 3.6.2 旧摘要字段与新摘要历史如何共存

现有三个字段继续保留，但角色要重新定义：

- `summary_Small`: 当前阶段的近期摘要聚合文本
- `summary_Week`: 按周压缩后的摘要聚合文本
- `summary_Backup`: 更早期摘要的兼容备份文本

同时新增结构化的 `summaryHistory[]`：

```javascript

        id: 'uuid',
        turnId: 'turn-001',
        week: currentWeek,
        summaryText: '第1年第1月第2周，主角在演武场练武，和呼延显切磋。',
```

两者关系：

1. `summaryHistory` 是 PromptBuilder 的直接输入
2. `summary_Small/Week/Backup` 是兼容旧模块、旧存档和旧展示逻辑的聚合缓存
3. 加载旧存档时如果没有 `summaryHistory`，可从 `summary_Backup + summary_Small` 尝试按空行切分做一次迁移恢复
4. 加载新存档时优先使用 `summaryHistory`，不再反向依赖三个旧字段拼接 prompt

#### 3.6.3 用存档内容重建完整 prompt

加载存档时，确实可以依据“存储的 `gameData` + 历史 `summaryHistory`”重建存档当时可用的 prompt 上下文：


这里的“完整 prompt”在 Phase 1 中定义为：

1. 5 个核心 system prompt 文件渲染结果
2. `<RecentHistory>` 块（来源于 `summaryHistory`）
3. 基于 `gameData` / `variableSystem` 生成的状态快照
4. 当前用户输入

它不要求恢复旧的完整正文历史，只要求恢复后续剧情继续推进所需的上下文。

### 3.7 摘要提取器

借鉴 TavernHeadless `core/generation/summary-extractor.ts`，增强现有 `parseOutput()`：

```javascript
// api/summary-extractor.js

const DEFAULT_TAG_NAMES = ['summary', 'SUMMARY', '摘要', 'memory', '记忆'];

function extractSummaries(text, tagNames = DEFAULT_TAG_NAMES) {
    const summaries = [];
    let cleanedText = text;

    for (const tag of tagNames) {
        const escaped = tag.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
        const regex = new RegExp(`<${escaped}>([\\s\\S]*?)<\\/${escaped}>`, 'gi');
        let match;
        while ((match = regex.exec(cleanedText)) !== null) {
            const content = match[1].trim();
            if (content) summaries.push(content);
        }
        cleanedText = cleanedText.replace(regex, '');
    }

    if (summaries.length > 0) cleanedText = cleanedText.replace(/\n{3,}/g, '\n\n').trim();
    return { summaries, cleanedText };
}
```

### 3.8 记忆注入选择器

借鉴 TavernHeadless 的 `MemoryInjectionSelector`，按记忆类型分配预算：

```javascript
// storage/memory-service.js（部分）

const MEMORY_TYPE_CONFIG = {
    fact: { budgetRatio: 0.35, maxItems: 10 },       // 结构化事实
    open_loop: { budgetRatio: 0.20, maxItems: 6 },   // 未解决的悬念
    summary: { budgetRatio: 0.45, maxItems: 5 },     // 回合摘要
};

function selectMemoriesForInjection(memories, tokenBudget, context) {
    const selected = [];

    for (const [type, config] of Object.entries(MEMORY_TYPE_CONFIG)) {
        const typeBudget = Math.floor(tokenBudget * config.budgetRatio);
        const candidates = memories
            .filter(m => m.type === type)
            .sort((a, b) => b.importance - a.importance)
            .slice(0, config.maxItems);

        let typeUsed = 0;
        for (const mem of candidates) {
            const tokens = Math.ceil(mem.content.length / 3);
            if (typeUsed + tokens > typeBudget) break;
            selected.push(mem);
            typeUsed += tokens;
        }
    }

    return selected;
}
```

#### 5.2.1 解析流程协调（extractSummaries 与 parseOutput 的关系）

现有 `index.html` 中的 `parseOutput()` 实际上是一个“混合函数”，只有 1 个调用点（页面初始化时 `parseOutput(outPut)`），内部同时承担了 4 类职责：

1. 调用 `removeThinkingContent()`，移除 `<think>` / `<thinking>` 类标签内容
2. 通过“首个汉字 / `#` / `*` 到首个 `<`”的规则提取正文，并直接写入 `localState.currentStoryText`
3. 提取 `<SUMMARY>`，按 `newWeek` 规则直接更新 `summary_Small` / `summary_Week` / `summary_Backup`，并依赖 `localState.summarySmallUpdateApplied` 做幂等保护
4. 从首个 `{` 到下一个 `<` 截取 JSON，`JSON.parse` 后直接调用 `parseLLMResponse()`，并依赖 `localState.turnUpdateApplied` 防止重复应用

这意味着当前 `parseOutput()` 并不是纯解析器，而是“解析 + 提交副作用”的组合。若直接在独立前端复用，会把 `SUMMARY` 累加、JSON 数值写入、正文显示三个阶段耦合在一起，不利于流水线错误恢复。

**改造原则：先拆，再兼容。** 独立前端新增纯解析入口；SillyTavern 旧逻辑保留兼容包装层，不在第一阶段破坏现有行为。

```javascript
// runtime/response-parser.js
class ResponseParser {
    run(rawText) {
        const cleaned = removeThinkingContent(rawText || '');
        const mainText = extractMainText(cleaned);
        const summaries = extractSummaries(cleaned);
        const sideNote = extractSideNoteJson(cleaned);

        return {
            raw: rawText,
            cleanedText: cleaned,
            mainText,
            summaries,
            sideNote,
        };
    }
}
```

其中各函数职责固定如下：

| 函数 | 是否纯函数 | 职责 |
|------|------------|------|
| `removeThinkingContent(text)` | 是 | 去掉 `<think>` / `<thinking>` 等思考标签 |
| `extractMainText(text)` | 是 | 复用现有“正文起点”规则，仅返回正文字符串 |
| `extractSummaries(text)` | 是 | 只负责提取 `<SUMMARY>` 内容数组，不写任何全局变量 |
| `extractSideNoteJson(text)` | 是 | 只负责找到 JSON 并 `JSON.parse`，失败时返回 `null` |
| `applySummaryUpdate(result, context)` | 否 | 按现有 `newWeek` 规则更新 `summary_Small/Week/Backup` |
| `applySideNote(result, context)` | 否 | 调用 `parseLLMResponse()` 或后续变量系统写入 |
| `applyParsedOutput(result, context)` | 否 | 统一提交正文、摘要、SIDE_NOTE |

**兼容方案：保留一个旧接口适配层。**

```javascript
// index.html / legacy-parser-bridge.js
function parseOutput(rawText, options = {}) {
    const parsed = responseParser.run(rawText);

    applyParsedOutput(parsed, {
        mode: options.mode || 'legacy',
        applySummary: options.applySummary !== false,
        applySideNote: options.applySideNote !== false,
        applyMainText: options.applyMainText !== false,
    });

    return parsed;
}
```

这样两种运行方式可以共存：

| 运行方式 | 调用方式 | 行为 |
|---------|---------|------|
| SillyTavern 旧页面初始化 / 恢复 | `parseOutput(outPut)` | 保持现状：解析后直接更新正文、摘要和数值 |
| 独立前端 Pipeline | `const parsed = responseParser.run(rawText)` | 先拿纯解析结果，再在提交阶段分别调用 `applySummaryUpdate` / `applySideNote` / `renderMainText` |

独立前端的提交流程应改为：

```javascript
// runtime/pipeline.js
const parsed = this.responseParser.run(rawText);

this.renderMainText(parsed.mainText);
applySummaryUpdate(parsed, { currentWeek, newWeek, localState, summaryVars });
applySideNote(parsed, { parseLLMResponse, variableSystem, localState });
```

这样做的原因：

1. 请求阶段拿到的只是 `parsed` 结果对象，没有副作用，失败后可直接重试
2. 提交阶段的副作用被拆成可控步骤，便于做 `rollbackTurn()`
3. ST 端仍可继续使用 `parseOutput(outPut)`，无需一次性重写整页逻辑

**关键约定更新如下：**

1. `ResponseParser.run()` 只做纯解析，不改全局状态
2. `extractSummaries` 不再内部写 `summary_Small`，摘要写入统一后移到提交阶段
3. `parseLLMResponse()` 不再由纯解析器直接调用，而是由 `applySideNote()` 或 Pipeline 提交阶段调用
4. 为兼容现有 ST 页面，保留 `parseOutput()` 作为包装函数，内部改为“纯解析 + 提交”的桥接实现

#### 5.2.2 错误恢复策略

Pipeline 的错误恢复分两段：

```javascript
async runTurn(message) {
    // === 请求阶段（无副作用，失败可安全重试）===
    const preprocessed = this.preprocessMessage(message);
    const messages = this.promptBuilder.buildMessages(...);

    let rawText;
    try {
        rawText = await this.apiService.sendMessages(messages);
    } catch (err) {
        // 请求失败：无需回滚任何状态
        showModal(`AI 请求失败：${err.message}\n\n可点击重试`);
        throw err;  // 让调用方知道失败了
    }

    // === 提交阶段（有副作用，但按顺序可控）===
    try {
        this.uiConversation.append({ role: 'user', content: preprocessed });
        const parsed = this.responseParser.run(rawText);
        this.variableSystem.set(parsed.sideNote);
        this.variableSystem.commitTurn();
        this.renderMainText(parsed.mainText);
        this.uiConversation.append({ role: 'assistant', content: parsed.mainText });
        await this.summaryHistory.append(parsed.summaries);
        await this.summaryHistory.trimWindow();
        await this.storageService.saveAppState({ gameData });
    } catch (err) {
        // 提交阶段失败（极少发生，通常是 IndexedDB 写入错误）
        // 回滚 turn 级变量，但不丢弃 AI 回复（已展示给用户）
        this.variableSystem.rollbackTurn();
        console.error('提交阶段错误，变量已回滚:', err);
        showModal('状态保存异常，游戏数值可能未更新。建议手动存档。');
    }
}
```

**设计要点：**
- 用户消息在 API 成功后才写入展示历史——失败时不会污染页面状态
- Prompt 历史只保存 `SUMMARY`，不会因旧正文过长快速挤爆上下文窗口
- 请求阶段失败不产生任何副作用，用户可以直接重试
- 提交阶段如果部分失败，回滚 turn 变量但保留已展示的文本

### 5.3 存档/读档/重新生成

```javascript
// 浏览器内存档
saveGame()
    → syncGameDataFromVariableSystem()
    → saveService.createSave(saveName, { gameData, summaryHistory, recentConversation })

// 导出 JSON 存档
exportSave()
    → syncGameDataFromVariableSystem()
    → buildSavePayload({ gameData, summaryHistory, recentConversation })
    → download as .json

// 浏览器内读档
loadGame(saveId)
    → saveService.loadSave(saveId)
    → variableSystem.importSession(saved.gameData)
    → restore summaryHistory
    → restore recentConversation
    → updateAllDisplays()

// 从 JSON 导入读档
importSave(file)
    → parse json
    → validate schema/version
    → variableSystem.importSession(payload.gameData)
    → restore summaryHistory
    → restore recentConversation
  → updateAllDisplays()

// 重新生成（Phase 1 即实现）
handleRegenerate()
  → variableSystem.rollbackTurn()          // 丢弃本轮变量
  → summaryHistory.popLast()               // 移除本轮写入的 SUMMARY
  → uiConversation.popLastAssistant()      // 移除 UI 中上条 AI 回复
  → syncGameDataFromVariableSystem()       // 恢复 gameData 到回滚后状态
  → rebuild prompt with last user message  // 用 lastUserMessage 重新构建 prompt
  → Pipeline.runTurn(lastUserMessage)      // 重走完整流水线
```

**Phase 1 就实现重新生成的理由：**

1. `variableSystem` 的 turn/session 分离已经提供了回滚基础设施
2. 重新生成是 LLM 游戏的核心体验保障——模型偶尔会输出格式错误或剧情偏差，没有重新生成等于没有纠错手段
3. 实现成本低：回滚 turnVars + 弹出最近一条 summary/UI 历史 + 重发即可

**注意事项：**

- 重新生成只回滚「最近一轮」，不支持连续多次回滚到更早状态（那是 Phase 2+ 的撤销栈）
- 如果上一轮已经 `commitTurn()` 了（即用户做了新操作后再想回滚上上轮），则不允许重新生成，按钮应置灰
- 重新生成期间 UI 应显示 loading 状态，防止用户重复点击

## 八、技术注意事项

### 8.1 CORS 跨域

**Phase 1：假设用户使用支持 CORS 的中转站。** 国内用户主流使用的 OpenAI 兼容中转站（如 new-api、one-api 类部署）通常已配置 `Access-Control-Allow-Origin: *`，直接从浏览器调用即可。Gemini API 原生支持浏览器直调。Phase 1 不提供任何代理组件，只在配置界面提示用户填入"支持 CORS 的 API 地址"。

**Phase 2：提供极简 Cloudflare Workers 代理脚本。** 对于需要直连 OpenAI 官方 API 的用户，提供一个 < 50 行的 Workers 脚本，部署后只做请求转发 + CORS 头注入，不缓存不记录。脚本放在 `scripts/cors-proxy-worker.js`，README 中附部署指南。

总结：

- Phase 1：零代理依赖，配置界面引导用户填入已支持 CORS 的地址
- Phase 2：可选的 Cloudflare Workers 极简代理
- Gemini：始终可直调，无需代理

### 8.2 API 密钥安全

- localStorage 中的密钥可被同域 XSS 读取
- 配置界面添加安全提示
- 密钥输入框使用 `type="password"`
- 不在 console 输出密钥

### 8.3 静态部署

- **Phase 1 可直接用 `file://` 打开**：提示词已硬编码、脚本用 `<script>` 标签加载、跨页数据用 URL 参数传递，均不依赖 `fetch` 本地文件
- **Phase 2+ 需 HTTP 服务器**：世界书条目改为 `fetch()` 加载 txt 文件后，`file://` 下 fetch 本地文件会被浏览器阻止
- 开发（Phase 2+）：`npx serve .` 或 VS Code Live Server
- 部署：GitHub Pages / Vercel / Netlify
- 打包：Capacitor (APK) / Tauri (EXE, ~10MB)

### 8.4 双模式兼容

所有改造点统一使用分支判断：

```javascript
const isStandalone = !getRenderFunction();

```

同一份代码可在 ST 中使用，也可独立部署。

### 8.7 parseOutput 拆分与兼容策略

根据现有实现，`parseOutput()` 不只是解析函数，还直接修改以下状态：

- `localState.currentStoryText`
- `summary_Small` / `summary_Week` / `summary_Backup`
- `parseLLMResponse(...)` 触发的数值与状态更新

因此不能简单理解为“`extractSummaries` 在前、`parseOutput` 在后”，而应拆成两层：

- 纯解析层：`removeThinkingContent`、`extractMainText`、`extractSummaries`、`extractSideNoteJson`
- 提交层：`applySummaryUpdate`、`applySideNote`、`renderMainText`

兼容策略如下：

1. 独立前端使用 `responseParser.run(rawText)`，只获取解析结果对象
2. ST 旧逻辑继续调用 `parseOutput(rawText)`，但其内部改为桥接到纯解析层，再执行提交层副作用
3. `parseOutput(rawText)` 仍保留返回值，便于后续逐步把旧调用点迁移到纯解析接口

桥接后，旧代码与新代码都共享同一套正文、SUMMARY、JSON 抽取规则，避免双份实现逐渐漂移。

### 8.8 Pipeline 错误恢复

Pipeline 采用"两阶段"设计确保错误安全：

| 阶段 | 包含步骤 | 失败时行为 |
|------|----------|------------|
| 请求阶段 | 预处理 → 构建 prompt → 调用 API | **无副作用**，直接报错，用户可重试 |
| 提交阶段 | 历史入库 → 解析 → 变量写入 → UI 更新 → 持久化 | 回滚 turn 变量，保留已展示文本，提示用户 |

关键设计：用户消息在 API 调用**成功后**才写入历史——失败时不会污染对话记录。

## 十、游戏入口改造：start-screen.html 作为独立前端入口

### 10.1 改造目标

将 `start-screen.html` 作为独立前端的入口页面。用户在此完成"新游戏创建"或"选择存档"，然后跳转到 `index.html` 进行游戏。`index.html` 不再允许无数据状态下直接访问（独立模式下自动跳回 start-screen）。

用户流程：

```
start-screen.html
  ├─ 江湖新篇 → 选难度 → 选出身 → 分配属性 → 确认 → 跳转 index.html（新游戏）
  ├─ 重续前缘 → 显示存档列表 → 选择存档 → 跳转 index.html（读档）
  └─ 武林快报 → 查看更新/反馈（不变）

index.html
  ├─ 检测 launchIntent → newGame / loadSave → 正常初始化
  ├─ 无 intent 但有 appState → 刷新恢复（不跳走）
  ├─ 无 intent 且无 appState（独立模式）→ 跳回 start-screen.html
  └─ 游戏内"返回主菜单"按钮 → 跳回 start-screen.html
```

### 10.2 两页面通信机制

使用 localStorage 中两个临时标记传递意图：

| Key | 值 | 用途 |
|-----|------|------|
| `jxz_launchIntent` | `'newGame'` 或 `'loadSave'` | 告诉 index.html 该走哪条初始化路径 |
| `jxz_loadSaveId` | 存档 ID 字符串 | 读档时指定加载哪个存档 |

**index.html 读取后立即清除这两个标记**，避免刷新时重复触发。

### 10.7 技术决策说明

**为什么用 localStorage 标记而不是 URL 参数？**
- start-screen 已依赖 localStorage 存 gameData，标记放在同一层更简单
- URL 参数在 `file://` 协议下行为不一致，且会被浏览器历史记录缓存

**为什么开场白在 start-screen 侧写入？**
- start-screen 拥有完整的角色创建上下文和 `outPut` 开场白文本
- 避免 index.html 需要再次持有开场白原文或知道角色创建细节

**为什么 index.html 有数据时刷新不跳走？**
- 用户在游戏中刷新页面是常见操作，应该无缝恢复
- 只有完全无数据（首次访问）才跳回 start-screen

##### 2.4.2 流式 SSE 请求实现

**api-service.js 新增**：

```js
// 流式发送（SSE）
async function sendMessagesStream(messages, callbacks) {
    // callbacks: { onToken(text), onComplete(fullText, usage), onError(err) }
    // 返回 AbortController 供外部中断
}
```

**OpenAI 兼容流式**：
- `POST /chat/completions`，`body.stream = true`
- `fetch()` 获取 `response.body`（ReadableStream）
- `TextDecoderStream` + 逐行解析 `data: {...}` 事件
- 提取 `choices[0].delta.content`，累加并调用 `callbacks.onToken(deltaText)`
- 收到 `data: [DONE]` 时调用 `callbacks.onComplete(fullText, usage)`
- 处理 `choices[0].delta.reasoning_content`（DeepSeek 思考链）：
  - 思考内容不渲染到正文，但累加保留用于调试日志
  - 只有 `content` 字段的 delta 才触发 `onToken`

**Gemini 流式**：
- `POST /models/{model}:streamGenerateContent?alt=sse&key=...`
- 响应格式：每个 SSE event 的 `data` 是完整 JSON 对象
- 提取 `candidates[0].content.parts[0].text`
- 同样通过 `callbacks.onToken` / `onComplete` 回调

**中断支持**：
- 创建 `AbortController`，将 `signal` 传入 `fetch()`
- `sendMessagesStream` 返回 `{ abort: () => controller.abort() }`
- 中断时触发 `callbacks.onComplete(partialText)`，已收到的内容仍然有效

**降级策略**：
- 流式请求失败（如 CORS 不支持 SSE、网络中断）时，自动回退到 `sendMessages()` 非流式调用
- 回退后通过 `callbacks.onComplete` 一次性返回完整内容

##### 2.4.3 Pipeline 流式集成

**pipeline.js 改造**：

`runTurn(message, options)` 改造为两阶段：

```
阶段 A（请求 + 流式渲染，无副作用）：
  1. 构建 messages（与现有逻辑相同）
  2. 调用 sendMessagesStream(messages, {
       onToken(delta) {

         var cleaned = responseParser.removeThinkingContent(accumulatedText);
         var mainText = responseParser.extractMainText(cleaned);
         if (mainText) renderMainText(mainText);
       },
       onComplete(fullText, usage) {

       },
       onError(err) {
         // 尝试回退非流式
       }

  1. responseParser.run(fullText) 完整解析

  4. syncGameDataFromVariables() + 持久化
  5. 处理事件
```

**关键设计决策**：
- 流式阶段（A）只做 UI 渲染，**不产生任何副作用**（不写 gameData、不存档、不更新摘要）
- 只有在 `onComplete` 拿到完整文本后才执行提交阶段（B），保证数据一致性
- 中断时：已收到的文本作为完整回复处理，仍然进入阶段 B（允许用户接受部分回复）

##### 2.4.4 流式渲染（复用现有链路，零改动 game-ui.js）

**核心架构**：流式渲染复用现有 `renderMainText` → `updateStoryText` → `updateStoryDisplay` 完整链路，不新增任何渲染函数。

**解析层安全性**：

流式期间 LLM 输出的 XML 标签可能未闭合（如 `<MAIN_TEXT>` 未关闭、`<SUMMARY>` 尚未出现），但 `responseParser.extractMainText()` 天然兼容此场景：

```js
// extractMainText 的匹配策略：
var match = text.match(/[#*\u4e00-\u9fff][^<]*/);  // 从第一个中文字符开始
var endIndex = text.indexOf('<', startIndex);         // 取到下一个 < 为止
// 若无 <（标签未闭合）→ endIndex=-1 → 取到字符串末尾 → 返回已有全部正文 ✅
```

| 流式阶段 | removeThinkingContent 后的文本 | extractMainText 结果 |
|---------|------------------------------|---------------------|
| thinking 未闭合 | `<ANAL>...部分思考` → 被截断为空 | `''`（不渲染，正确） |
| thinking 结束，正文开始 | `正文第一段\|npc\|...\n正文第二` | 返回已有全部正文 ✅ |
| 正文继续增长 | 更多段落累积 | 返回更多正文 ✅ |
| `</MAIN_TEXT>` 到达 | `...正文最后段\n</MAIN_TEXT>` | 取到 `<` 停止 → 完整正文 ✅ |
| SUMMARY 到达 | `...正文\n</MAIN_TEXT>\n<SUMMARY>...` | 仍取到首个闭合 `<` → 正文不受污染 ✅ |

**渲染层安全性**：

`updateStoryText` 是纯渲染函数（无累积状态），每次调用都从头重建 `storyPages` 数组。ST 链路的流式效果就是通过每个 token 到达后反复调用此函数实现的，证明它天然支持高频调用。独立前端的流式调用方与 ST 链路完全隔离（来自 `pipeline.js`，ST 不加载），不会交叉影响。

**流式分页行为**：
- 流式开始时：自动跳转到第 1 页（`currentPage = 0`）
- 随着文本增长：新段落产生新页码，页码指示器动态更新
- 用户可自由翻页查看已完成的页码
- `newWeek===1` 时保持全文展开模式
- GameMode===1（SLG模式）时：已完成的页码的背景、角色差分、CG 正常渲染（由 `updateStoryDisplay` 中的 SLG 图层逻辑处理）

**节流优化**（避免每个 token 都重建 DOM）：

```js
// pipeline.js 中的流式回调
var _streamThrottleTimer = null;
var _streamAccText = '';

function onStreamToken(delta) {
    _streamAccText += delta;
    if (!_streamThrottleTimer) {
        _streamThrottleTimer = setTimeout(function() {
            _streamThrottleTimer = null;
            var cleaned = responseParser.removeThinkingContent(_streamAccText);
            var mainText = responseParser.extractMainText(cleaned);
            if (mainText) renderMainText(mainText);
        }, 100);  // 每 100ms 最多渲染一次
    }
}

function onStreamComplete(fullText) {
    clearTimeout(_streamThrottleTimer);
    _streamThrottleTimer = null;
    // 阶段 B：完整解析 + 最终渲染
    var parsed = responseParser.run(fullText);
    renderMainText(parsed.mainText);
    // ... 后续提交逻辑
}
```

**thinking/推理内容过滤**：
- DeepSeek `reasoning_content`：在 SSE 解析层直接过滤，不累加到 `accumulatedText`，单独收集用于调试
- `<ANAL>` 等标签内容：由 `responseParser.removeThinkingContent()` 在每次渲染前过滤（含未闭合标签截断）
- 最终完整文本由 `responseParser.run()` 统一处理，与非流式链路完全一致

#### 2.5 CORS 代理方案

**背景**：游戏部署到 GitHub Pages 等线上环境后，浏览器会强制执行 CORS 策略，直接请求第三方 LLM API 会被拦截报错。本地 `file://` 运行、Electron 打包（EXE）、Android WebView 打包（APK）均不受此限制，只有线上网页部署需要代理。

**架构**：
```
api-service.js
  └─ _resolveUrl(targetUrl)
       ├─ file:// / Electron / WebView → 直连 targetUrl
       └─ 网页部署(web) → corsProxyUrl + "?target=" + encodeURIComponent(targetUrl)

cors-proxy（Cloudflare Worker）
  └─ 收到请求 → 读取 ?target 参数 → 转发请求 → 加 CORS 头 → 返回
```

**安全原则**：
- API Key 永远由用户浏览器端持有，通过 `Authorization` header 透传，代理不存储不记录
- Worker 限制 `target` 参数只允许 HTTPS URL，防止被滥用作开放代理
- Worker 纯粹做 CORS 头注入和流量中转，无任何身份验证逻辑

**实现清单**：
